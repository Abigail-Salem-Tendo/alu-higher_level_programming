MyList = __import__('1-my_list').MyList

# Create an instance
my_list = MyList()
my_list.append(4)
my_list.append(1)
my_list.append(5)
my_list.append(3)
my_list.append(2)

# Original list should be in original order
print(my_list)
[4, 1, 5, 3, 2]

# Sorted print
my_list.print_sorted()
[1, 2, 3, 4, 5]

# Original list remains unchanged
print(my_list)
[4, 1, 5, 3, 2]
